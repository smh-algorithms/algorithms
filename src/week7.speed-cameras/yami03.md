function solution(routes) {
  let count = 1;
  
  routes.sort((a, b) => a[0] - b[0]);
  let minEndPoint = routes[0][1];
  
  for (let i = 1; i < routes.length; i++) {
    if (minEndPoint < routes[i][0]) {
      count++;
      minEndPoint = routes[i][1]
    } else {
      minEndPoint = Math.min(minEndPoint, routes[i][1])
    }
  }

  return count;
}
